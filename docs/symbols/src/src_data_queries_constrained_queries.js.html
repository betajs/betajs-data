 33 	emulate: function (constrained_query, query_capabilities, query_function, query_context, callbacks) {
 34 		var query = constrained_query.query || {};
 35 		var options = constrained_query.options || {};
 36 		var execute_query = {};
 37 		var execute_options = {};
 38 		if ("sort" in options && "sort" in query_capabilities)
 39 			execute_options.sort = options.sort;
 40 		// Test
 41 		execute_query = query;
 42 		if ("query" in query_capabilities || BetaJS.Types.is_empty(query)) {
 43 			execute_query = query;
 44 			if (!options.sort || ("sort" in query_capabilities)) {
 45 				if ("skip" in options && "skip" in query_capabilities)
 46 					execute_options.skip = options.skip;
 47 				if ("limit" in options && "limit" in query_capabilities) {
 48 					execute_options.limit = options.limit;
 49 					if ("skip" in options && !("skip" in query_capabilities))
 50 						execute_options.limit += options.skip;
 51 				}
 52 			}
 53 		}  
 54 		var params = [execute_query, execute_options];
 55 		if (callbacks)
 56 			params.push(callbacks);
 57 		var success_call = function (raw) {
 58 			var iter = raw;
 59 			if (raw === null)
 60 				iter = new BetaJS.Iterators.ArrayIterator([]);
 61 			else if (BetaJS.Types.is_array(raw))
 62 				iter = new BetaJS.Iterators.ArrayIterator(raw);		
 63 			if (!("query" in query_capabilities || BetaJS.Types.is_empty(query)))
 64 				iter = new BetaJS.Iterators.FilteredIterator(iter, function(row) {
 65 					return BetaJS.Queries.evaluate(query, row);
 66 				});
 67 			if ("sort" in options && !("sort" in execute_options))
 68 				iter = new BetaJS.Iterators.SortedIterator(iter, BetaJS.Comparators.byObject(options.sort));
 69 			if ("skip" in options && !("skip" in execute_options))
 70 				iter = new BetaJS.Iterators.SkipIterator(iter, options["skip"]);
 71 			if ("limit" in options && !("limit" in execute_options))
 72 				iter = new BetaJS.Iterators.LimitIterator(iter, options["limit"]);
 73 			if (callbacks && callbacks.success)
 74 				BetaJS.SyncAsync.callback(callbacks, "success", iter);
 75 			return iter;
 76 		};
 77 		var exception_call = function (e) {
 78 			if (callbacks && callbacks.exception)
 79 				BetaJS.SyncAsync.callback(callbacks, "exception", e);
 80 			else
 81 				throw e;
 82 		};
 83 		if (callbacks) 
 84 			query_function.apply(query_context || this, [execute_query, execute_options, {
 85 				success: success_call,
 86 				exception: exception_call,
 87 				sync: callbacks.sync,
 88 				context: callbacks.context || this
 89 			}]);
 90 		else
 91 			try {
 92 				var raw = query_function.apply(query_context || this, [execute_query, execute_options]);
 93 				return success_call(raw);
 94 			} catch (e) {
 95 				exception_call(e);
 96 			}
 97 		return true;	
 98 	},

100 	subsumizes: function (query, query2) {
101 		qopt = query.options || {};
102 		qopt2 = query2.options || {};
103 		qskip = qopt.skip || 0;
104 		qskip2 = qopt2.skip || 0;
105 		qlimit = qopt.limit || null;
106 		qlimit2 = qopt2.limit || null;
107 		qsort = qopt.sort;
108 		qsort2 = qopt2.sort;
109 		if (qskip > qskip2)
110 			return false;
111 		if (qlimit) {
112 			if (!qlimit2)
113 				return false;
114 			if (qlimit2 + qskip2 > qlimit + qskip)
115 				return false;
116 		}
117 		if ((qskip || qlimit) && (qsort || qsort2) && JSON.stringify(qsort) != JSON.stringify(qsort2))
118 			return false;
119 		return BetaJS.Queries.subsumizes(query.query, query2.query);
120 	},

132 			return false;
133 		var qots = query.options || {};
134 		var qopts2 = query2.options || {};
135 		if (JSON.stringify(qopts.sort || {}) != JSON.stringify(qopts2.sort || {}))
136 			return false;
137 		if ("skip" in qopts) {
138 			if ("skip" in qopts2) {
139 				if (qopts.skip <= qopts2.skip)
140 					return !qopts.limit || (qopts.skip + qopts.limit >= qopts2.skip);
141 				else
142 					return !qopts2.limit || (qopts2.skip + qopts2.limit >= qopts.skip);
143 			} else 
144 				return (!qopts2.limit || (qopts2.limit >= qopts.skip));
145 		} else 
146 			return !("skip" in qopts2) || (!qopts.limit || (qopts.limit >= qopts2.skip));
147 	},

156 				limit: "limit" in qopts ? ("limit" in qopts2 ? Math.max(qopts.limit, qopts2.limit): null) : null,
157 				sort: query.sort
