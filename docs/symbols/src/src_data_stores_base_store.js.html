 33 /** @class */
 34 BetaJS.Stores.BaseStore = BetaJS.Stores.ListenerStore.extend("BetaJS.Stores.BaseStore", [
 35 	BetaJS.SyncAsync.SyncAsyncMixin,
 36 	/** @lends BetaJS.Stores.BaseStore.prototype */
 37 	{
 38 		
 39 	constructor: function (options) {
 40 		this._inherited(BetaJS.Stores.BaseStore, "constructor", options);
 41 		options = options || {};
 42 		this._id_key = options.id_key || "id";
 43 		this._create_ids = options.create_ids || false;
 44 		this._last_id = 1;
 45 		this._supportsSync = true;
 46 		this._supportsAsync = true;
 47 		this._query_model = "query_model" in options ? options.query_model : null;
 48 	},
 49 	
 50     query_model: function () {
 51         if (arguments.length > 0)
 52             this._query_model = arguments[0];
 53         return this._query_model;
 54     },
 55     
 56 	/** Insert data to store. Return inserted data with id.
 57 	 * 
 58  	 * @param data data to be inserted
 59  	 * @return data that has been inserted with id.
 60  	 * @exception if it fails
 61 	 */
 62 	_insert: function (data, callbacks) {
 63 		throw new BetaJS.Stores.StoreException("unsupported: insert");
 64 	},
 65 	
 66 	/** Remove data from store. Return removed data.
 67 	 * 
 68  	 * @param id data id
 69  	 * @exception if it fails
 70 	 */
 71 	_remove: function (id, callbacks) {
 72 		throw new BetaJS.Stores.StoreException("unsupported: remove");
 73 	},
 74 	
 75 	/** Get data from store by id.
 76 	 * 
 77 	 * @param id data id
 78 	 * @return data
 79 	 * @exception if it fails
 80 	 */
 81 	_get: function (id, callbacks) {
 82 		throw new BetaJS.Stores.StoreException("unsupported: get");
 83 	},
 84 	
 85 	/** Update data by id.
 86 	 * 
 87 	 * @param id data id
 88 	 * @param data updated data
 89 	 * @return data from store
 90 	 * @exception if it fails
 91 	 */
 92 	_update: function (id, data, callbacks) {
 93 		throw new BetaJS.Stores.StoreException("unsupported: update");
 94 	},
 95 	
 96 	_query_capabilities: function () {
 97 		return {};
 98 	},
 99 	
100 	/*
101 	 * @exception if it fails
102 	 */
103 	_query: function (query, options, callbacks) {
104 		throw new BetaJS.Stores.StoreException("unsupported: query");
105 	},
106 	
107 	insert: function (data, callbacks) {
108 		var event_data = null;
109 		if (BetaJS.Types.is_array(data)) {
110 			event_data = data[1];
111 			data = data[0];
112 		}			
113 		if (this._create_ids && !(this._id_key in data && data[this._id_key])) {
114 			while (this.get(this._last_id))
115 				this._last_id++;
116 			data[this._id_key] = this._last_id;
117 		}
118 		return this.then(this._insert, [data], callbacks, function (row, callbacks) {
119 			this._inserted(row, event_data);
120 			BetaJS.SyncAsync.callback(callbacks, "success", row);
121 		});
122 	},
123 	
124 	insert_all: function (data, callbacks, query) {
125 		var event_data = null;
126 		if (arguments.length > 3)
127 			event_data = arguments[3];
128 		if (query && this._query_model) {
129 			this.trigger("query_register", query);
130 			this._query_model.register(query);
131 		}
132 		if (callbacks) {
133 			var self = this;
134 			var f = function (i) {
135 				if (i >= data.length) {
136 					BetaJS.SyncAsync.callback(callbacks, "success");
137 					return;
138 				}
139 				this.insert(event_data ? [data[i], event_data] : data[i], BetaJS.SyncAsync.mapSuccess(callbacks, function () {
140 					f.call(self, i + 1);
141 				}));
142 			};
143 			f.call(this, 0);
144 		} else {
145 			for (var i = 0; i < data.length; ++i)
146 				this.insert(event_data ? [data[i], event_data] : data[i]);
147 		}
148 	},
149 
150 	remove: function (id, callbacks) {
151 		var event_data = null;
152 		if (BetaJS.Types.is_array(id)) {
153 			event_data = id[1];
154 			id = id[0];
155 		}			
156 		return this.then(this._remove, [id], callbacks, function (result, callbacks) {
157 			this._removed(id, event_data);
158 			BetaJS.SyncAsync.callback(callbacks, "success", id);
159 		});
160 	},
161 	
162 	get: function (id, callbacks) {
163 		return this.delegate(this._get, [id], callbacks);
164 	},
165 	
166 	update: function (id, data, callbacks) {
167 		var event_data = null;
168 		if (BetaJS.Types.is_array(data)) {
169 			event_data = data[1];
170 			data = data[0];
171 		}			
172 		return this.then(this._update, [id, data], callbacks, function (row, callbacks) {
173 			this._updated(row, data, event_data);
174 			BetaJS.SyncAsync.callback(callbacks, "success", row, data);
175 		});
176 	},
177 	
178 	query: function (query, options, callbacks) {
179 		if (options) {
180 			if (options.limit)
181 				options.limit = parseInt(options.limit, 10);
182 			if (options.skip)
183 				options.skip = parseInt(options.skip, 10);
184 		}
185 		if (this._query_model) {
186 		    var subsumizer = this._query_model.subsumizer_of({query: query, options: options});
187     		if (!subsumizer) {
188     			this.trigger("query_miss", {query: query, options: options});
189     			var e = new BetaJS.Stores.StoreException("Cannot execute query");
190     			if (callbacks)
191     			    BetaJS.SyncAsync.callback(callbacks, "exception", e);
192     			else
193     				throw e;
194     			return null;
195     		} else
196     		    this.trigger("query_hit", {query: query, options: options}, subsumizer);
197 		}
198 		var q = function (callbacks) {
199 			return BetaJS.Queries.Constrained.emulate(
200 				BetaJS.Queries.Constrained.make(query, options || {}),
201 				this._query_capabilities(),
202 				this._query,
203 				this,
204 				callbacks);			
205 		};
206 		return this.either(callbacks, q, q);
207 	},
208 	
209 	_query_applies_to_id: function (query, id) {
210 		var row = this.get(id);
211 		return row && BetaJS.Queries.overloaded_evaluate(query, row);
212 	},
213 	
214 	clear: function (callbacks) {
215 		return this.then(this.query, [{}, {}], callbacks, function (iter, callbacks) {
216 			var promises = [];
217 			while (iter.hasNext())
218 				promises.push(this.remove, [iter.next().id]);
219 			return this.join(promises, callbacks);
220 		});
221 	},
222 	
223 	_ensure_index: function (key) {
224 	},
225 	
226 	ensure_index: function (key) {
227 		this._ensure_index(key);
228 	},
229 	
230 	perform: function (commit, callbacks) {
231 		var action = BetaJS.Objs.keyByIndex(commit);
232 		var data = BetaJS.Objs.valueByIndex(commit);
233 		if (action == "insert")
234 			this.insert(data, callbacks);
235 		else if (action == "remove")
236 			this.remove(data, callbacks);
237 		else if (action == "update")
238 			this.update(BetaJS.Objs.keyByIndex(data), BetaJS.Objs.valueByIndex(data), callbacks);
239 		else
240 			throw new BetaJS.Stores.StoreException("unsupported: perform " + action);
241 	},
242 	
243 	bulk: function (commits, optimistic, callbacks) {
244 		var result = [];
245 		if (callbacks) {
246 			var helper = function () {
247 				if (result.length < commits.length) {
248 					this.perform(commits[result.length], {
249 						context: this,
250 						success: function () {
251 							result.push(true);
252 							helper.apply(this);
253 						},
254 						exception: function (e) {
255 							result.push(false);
256 							if (optimistic)
257 								helper.apply(this);
258 							else
259 								callbacks.exception.apply(callbacks.context || this, e);
260 						}
261 					});
262 				} else
263 					callbacks.success.call(callbacks.context || this, result);
264 			};
265 			helper.apply(this);
266 		} else {
267 			for (var i = 0; i < commits.length; ++i) {
268 				try {
269 					this.perform(commits[i]);
270 					result.push(true);
271 				} catch (e) {
272 					result.push(false);
273 					if (!optimistic)
274 						throw e;
275 				}
276 			}
277 		}
278 		return result;
279 	}
